( function ( window ){
  "use strict";
  var
    wakegi = window.wakegi,
    Browser = wakegi.Browser;

  Browser.CriOS = ( function (){
    var
      versions = [],
      crios, version, major;

    /**
     * iOS Chrome 判定
     * @class CriOS
     * @constructor
     */
    function CriOS () {
      throw new Error( "CriOS can't create instance." );
    }

    var p = CriOS.prototype;

    p.constructor = CriOS;

    /**
     * @method init
     * @static
     */
    CriOS.init = function () {

      if ( typeof crios === "undefined" ) {

        crios = !!Browser.ua().match(/crios/i);

      }

    };

    /**
     * @method calculate
     * @static
     */
    CriOS.calculate = function () {
      var
        numbers, int, i, limit;

      if ( typeof version === "undefined" ) {
        // version undefined
        numbers = Browser.app().match(/CriOS\/(\d+)\.(\d+)\.(\d+)\.?(\d+)?/);
        version = "";
        major = -1;

        if ( Array.isArray( numbers ) ) {
          // 結果が配列
          int = wakegi.int;

          for ( i = 1, limit = numbers.length; i < limit; i++ ) {

            versions.push( int( numbers[ i ], 10 ) );

          }

          version = versions.join( "." );
          major = versions[ 0 ];

        }

      }

    };

    /**
     * @method is
     * @static
     * @return {boolean}
     */
    CriOS.is = function () {

      CriOS.init();
      return crios;

    };

    /**
     *
     * @method version
     * @static
     * @return {string} NN.NN.NN.NN 型（文字）で返します
     */
    CriOS.version = function () {

      CriOS.calculate();
      return version;

    };

    /**
     * @method
     * @static
     * @return {int}
     */
    CriOS.major = function () {

      CriOS.calculate();
      return major;

    };

    return CriOS;
  }() );
}( window ) );
