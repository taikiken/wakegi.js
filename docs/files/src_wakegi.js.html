var wakegi = wakegi || {};

wakegi.int = parseInt;
wakegi.float = parseFloat;

( function (){
  "use strict";

  // Array.isArray
  // https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Array/isArray
  if ( !Array.isArray ) {

    Array.isArray = function( arg ) {

      return Object.prototype.toString.call( arg ) === '[object Array]';
    };
  }

}() );
